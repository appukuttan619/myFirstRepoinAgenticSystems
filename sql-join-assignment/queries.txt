CREATE TABLE users (
user_id INTEGER PRIMARY KEY,
name TEXT,
signup_date DATE
);

CREATE TABLE orders (
order_id INTEGER PRIMARY KEY,
user_id INTEGER,
amount INTEGER
);

CREATE TABLE activity (
user_id INTEGER,
login_count INTEGER
);

Question 3
SELECT u.name, o.amount
FROM users u
INNER JOIN orders o
ON u.user_id = o.user_id;

Question 4
SELECT u.name, SUM(o.amount) AS total_order_amount
FROM users u
INNER JOIN orders o
ON u.user_id = o.user_id
GROUP BY u.name;

Question 5
SELECT u.name, a.login_count
FROM users u
LEFT JOIN activity a
ON u.user_id = a.user_id;

Question 6
SELECT u.*
FROM users u
LEFT JOIN orders o
ON u.user_id = o.user_id
WHERE o.order_id IS NULL;

Question 7
SELECT u.user_id, u.name, SUM(o.amount) AS total_order_amount, a.login_count
FROM users u
LEFT JOIN orders o
ON u.user_id = o.user_id
LEFT JOIN activity a
ON u.user_id = a.user_id
GROUP BY u.user_id, u.name, a.login_count;

Question 8

ON u.user_id = o.user_id
WHERE o.amount > 400;

Question 9
SELECT u.name, SUM(o.amount) AS total_order_amount
FROM users u
INNER JOIN orders o
ON u.user_id = o.user_id
GROUP BY u.name
ORDER BY total_order_amount DESC
LIMIT 3;

Question 10
SELECT u.name, SUM(o.amount) AS total_order_amount, a.login_count
FROM users u
LEFT JOIN orders o
ON u.user_id = o.user_id
LEFT JOIN activity a
ON u.user_id = a.user_id
GROUP BY u.name, a.login_count
ORDER BY total_order_amount DESC, a.login_count DESC;

Question 11
SELECT u.name, AVG(o.amount) AS average_order_amount
FROM users u
INNER JOIN orders o
ON u.user_id = o.user_id
GROUP BY u.name;

Question 12
SELECT u.user_id, u.name, SUM(o.amount) AS total_order_amount, a.login_count
FROM users u
LEFT JOIN orders o
ON u.user_id = o.user_id
LEFT JOIN activity a
ON u.user_id = a.user_id
GROUP BY u.user_id, u.name, a.login_count
ORDER BY total_order_amount DESC
LIMIT 3;
